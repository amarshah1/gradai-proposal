\centering
\tikzset{
  box/.style={rectangle, rounded corners=3pt, draw, align=center,
              minimum width=2.6cm, minimum height=0.85cm, font=\normalsize},
  goldbox/.style={box, fill=yellow!50, draw=orange!90!black, very thick},
  genbox/.style={box, fill=blue!30, draw=blue!80!black, very thick},
  rlbox/.style={box, fill=green!20, draw=green!60!black, very thick},
  evalbox/.style={box, fill=red!30, draw=red!80!black, very thick},
  arr/.style={-{Stealth[length=5pt]}, thick},
}
\begin{tikzpicture}[node distance=0.55cm and 1.45cm]

  % Top: benchmark input
  \node[goldbox, minimum width=4.2cm] (bench)
    {Benchmark\\{\small NL spec + gold Verus spec}};

  % NL spec fed to generator
  \node[rlbox, below=of bench] (gen)
    {LLM Generator\\{\small (given NL spec only)}};
  \draw[arr] (bench) -- node[right, font=\small]{NL spec} (gen);

  % Two branches
  \node[genbox, below left=1.0cm and 1.8cm of gen]  (tests)
    {Unit Tests};
  \node[genbox, below right=1.0cm and 1.8cm of gen] (spec)
    {Verus Spec\\{\small (generated)}};
  \draw[arr] (gen) -- (tests);
  \draw[arr] (gen) -- (spec);

  % RL training nodes
  \node[rlbox, below=0.9cm of tests] (rlA)
    {LLM Implementor\\{\small goal: pass unit tests}};
  \node[rlbox, below=0.9cm of spec]  (rlB)
    {LLM Implementor\\{\small goal: satisfy Verus spec}};
  \draw[arr] (tests) -- (rlA);
  \draw[arr] (spec)  -- (rlB);

  % Code outputs
  \node[box, below=0.75cm of rlA] (codeA) {Code A};
  \node[box, below=0.75cm of rlB] (codeB) {Code B};
  \draw[arr] (rlA) -- (codeA);
  \draw[arr] (rlB) -- (codeB);

  % Evaluation node
  \node[evalbox, minimum width=4.2cm,
        below=0.75cm of $(codeA)!0.5!(codeB)$] (eval)
    {Evaluate \\{\small reward hacking $\Leftrightarrow$ gold Verus spec not satisfied}};
  \draw[arr] (codeA) -- (eval);
  \draw[arr] (codeB) -- (eval);

  % Gold spec dashed arrow going around the right side
  \draw[arr, dashed, draw=orange!80!red!90!black, very thick] (bench.east)
        -- ++(5.0,0)
        |- (eval.east)
        node[pos=0.25, right, font=\normalsize\bfseries, text=orange!80!red!90!black]{gold spec};

\end{tikzpicture}
\caption{Experimental setup. Both branches receive only the natural language specification as input. The dashed arrow indicates the gold-standard Verus specification used solely for final evaluation.}
\label{fig:setup}
